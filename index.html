Quick Start

Get up and running with Polaris in just a few lines of code.

Python 
from polaris.environments import StrategicExperimentationEnv
from polaris.agents import TD3Agent
from polaris.training import MultiAgentTrainer

# Initialize environment with complete network topology
env = StrategicExperimentationEnv(
    num_agents=2,
    network_type="complete",
    strategic_params={"exploration_bonus": 0.1}
)

# Create agents with TD3 algorithm
agents = [TD3Agent(env.observation_space, env.action_space) 
          for _ in range(env.num_agents)]

# Initialize trainer
trainer = MultiAgentTrainer(env, agents)

# Train the multi-agent system
results = trainer.train(total_episodes=10000)

# Evaluate performance
evaluation_results = trainer.evaluate(num_episodes=100)
print(f"Average reward: {evaluation_results['mean_reward']}")
